schoolData[1] : "Sekolah Tidak Ditemukan" }; } return { isLoggedIn: false }; } function logout() { PropertiesService.getUserProperties().deleteAllProperties(); return { status: "sukses" }; } // ================================================================= // FUNGSI OTENTIKASI & REGISTRASI (DENGAN HASHING) // ================================================================= function registerSchoolAndAdmin(data) { try { const users = getSheetData(sheets.users); const schools = getSheetData(sheets.schools); if (users.some(u => u[0] === data.adminEmail)) return { status: "gagal", message: "Email ini sudah terdaftar." }; if (schools.some(s => s[2] === data.npsn)) return { status: "gagal", message: "NPSN sekolah ini sudah terdaftar." }; const schoolId = "SCH-" + new Date().getTime(); sheets.schools.appendRow([schoolId, data.namaSekolah, data.npsn, data.jenjang, data.alamat, data.namaKepsek, data.nipKepsek]); const otp = generateOtp(); const expiry = new Date().getTime() + (5 * 60 * 1000); const passwordHash = hashPassword(data.password); sheets.users.appendRow([data.adminEmail, passwordHash, data.adminNama, schoolId, "Admin", otp, expiry, "PENDING"]); sheets.pengaturan.appendRow([generateUniqueId(), schoolId, "Bobot Ujian Madrasah (%)", 40]); sheets.pengaturan.appendRow([generateUniqueId(), schoolId, "Bobot Nilai Rapor (%)", 60]); sendOtpEmail(data.adminEmail, otp); return { status: "sukses", email: data.adminEmail, message: "Pendaftaran sekolah berhasil. Silakan login." }; } catch (e) { return { status: "gagal", message: "Terjadi kesalahan server: " + e.message }; } } function userLogin(credentials) { const users = getSheetData(sheets.users); const user = users.find(u => u[0] === credentials.email); if (!user) return { status: "gagal", message: "Email atau password salah." }; const storedHash = user[1]; if (storedHash !== hashPassword(credentials.password)) return { status: "gagal", message: "Email atau password salah." }; if (user[7] !== "TERVERIFIKASI") return { status: "gagal", message: "Akun Anda belum diverifikasi." }; const userProps = PropertiesService.getUserProperties(); userProps.setProperties({ 'isLoggedIn': 'true', 'email': user[0], 'schoolId': user[3] }); return { status: "sukses" }; } // ================================================================= // FUNGSI CRUD & IMPORT/EXPORT // ================================================================= function getSiswaDataForSchool() { return getMySchoolData('siswa'); } function getMySchoolData(dataType) { try { const userProps = PropertiesService.getUserProperties(); const schoolId = userProps.getProperty('schoolId'); if (!schoolId) return { status: 'gagal', message: 'Sesi tidak valid. const sheet = sheets.templateSiswa; if (!sheet) throw new Error("Sheet 'TemplateSiswa' tidak ditemukan."); const sheetId = sheet.getSheetId(); return `https://docs.google.com/spreadsheets/d/${spreadsheetId}/export?format=xlsx&gid=${sheetId}`; } function importSiswaFromExcel(fileData) { const schoolId = PropertiesService.getUserProperties().getProperty('schoolId'); if (!schoolId) return { status: 'gagal', message: 'Sesi tidak valid.' }; try { const blob = Utilities.newBlob(Utilities.base64Decode(fileData.data), fileData.mimeType, fileData.fileName); const data = sheet.getDataRange().getValues(); data.shift(); let importedCount = 0, skippedCount = 0; const siswaToAppend = data.reduce((acc, row) => { const [nisn, nama, nisLokal, jk, tempatLahir, tglLahir, alamat, thnMasuk, thnLulus] = row; if (!nisn || !nama) { skippedCount++; } else { importedCount++; acc.push([ generateUniqueId(), schoolId, nisn, nama, nisLokal || '', jk || '', tempatLahir || '', tglLahir || null, alamat || '', thnMasuk || '', thnLulus || '' ]); } return acc; }, []); if (siswaToAppend.length > 0) { sheets.siswa.getRange(sheets.siswa.getLastRow() + 1, 1, siswaToAppend.length, siswaToAppend[0].length).setValues(siswaToAppend); } Drive.Files.remove(tempSheetFile.id); let message = `Impor Selesai. ${importedCount} siswa berhasil diimpor.`; if (skippedCount > 0) message += ` ${skippedCount} baris dilewati karena NISN/Nama kosong.`; return { status: 'sukses', message: message }; } catch (e) { return { status: 'gagal', message: `Terjadi kesalahan saat impor: ${e.message}` }; } } // ================================================================= // FUNGSI HELPERS // ================================================================= function getSheetData(sheet) { if (!sheet || sheet.getLastRow() < 2) return []; ${otp}

` }); } function hashPassword(password) { const digest = Utilities.computeDigest(Utilities.DigestAlgorithm.SHA_256, password); return Utilities.bytesToHex(digest); }
